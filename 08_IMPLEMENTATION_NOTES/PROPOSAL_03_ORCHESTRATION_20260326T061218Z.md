import argparse
import json
import os
from pathlib import Path


class CorePromptBuilder:
    def interactive_build(self) -> dict:
        print("Initializing core prompt builder interactively...")
        domain = input("Enter Domain (e.g., 03_ORCHESTRATION): ").strip() or "03_ORCHESTRATION"
        mtype = input("Enter Type (e.g., IMPLEMENTATION): ").strip() or "IMPLEMENTATION"
        deliverable = input("Enter Deliverable description: ").strip()
        requirements = input("Enter Requirements details: ").strip()
        
        config = {
            "domain": domain,
            "type": mtype,
            "deliverable": deliverable,
            "requirements": requirements
        }
        print(f"Config built: {config}")
        return config

    def generate_mission_file(self, config: dict, output_dir: Path) -> Path:
        os.makedirs(output_dir, exist_ok=True)
        mission_name = f"{config['domain']}_{config['type']}.md"
        mission_path = output_dir / mission_name
        
        content = f"""# MISSION BRIEF

## Domain
{config['domain']}

## Type
{config['type']}

## Deliverable
{config['deliverable']}

## Requirements
{config['requirements']}

---
*Generated by CorePromptBuilder*
"""
        
        with open(mission_path, 'w') as f:
            f.write(content)
        
        return mission_path

    def queue_mission(self, mission_path: Path, queue_path: Path, priority: int) -> None:
        os.makedirs(os.path.dirname(queue_path), exist_ok=True)
        
        if not queue_path.exists():
            with open(queue_path, 'w') as f:
                json.dump([], f)
        
        # Load existing queue
        queue = []
        try:
            with open(queue_path, 'r') as f:
                queue = json.load(f)
        except (json.JSONDecodeError, FileNotFoundError):
            queue = []
            
        queue.append({
            "id": str(mission_path.name),
            "path": str(mission_path.absolute()),
            "priority": priority
        })
        
        with open(queue_path, 'w') as f:
            json.dump(queue, f, indent=2)

    def validate_mission(self, path: Path) -> list:
        errors = []
        if not path.exists():
            return ["Error: Mission file does not exist."]
        
        with open(path, 'r') as f:
            content = f.read()
            
        if "# MISSION BRIEF" not in content:
            errors.append("Validation Failed: Missing main header '# MISSION BRIEF'.")
            
        if not config := path.with_suffix('') or path.stem:
            # Simple check to ensure it looks like a generated file isn't strictly required 
            # but we return errors list.
            pass
            
        return errors


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Core Prompt CLI")
    parser.add_argument("--domain", type=str, default=None)
    parser.add_argument("--type", type=str, default=None)
    parser.add_argument("--output-dir", type=Path, default=Path("./missions"))
    parser.add_argument("--queue-path", type=Path, default=Path("./MISSION_QUEUE.json"))
    parser.add_argument("--priority", type=int, default=0)
    
    args = parser.parse_args()
    
    builder = CorePromptBuilder()
    
    config = None
    
    if args.domain is None and args.type is None:
        # If arguments omitted, prompt interactively
        print("No domain or type provided. Running interactive build.")
        config = builder.interactive_build()
        # Use defaults from interactive step or args if partially set?
        # Brief says "If arguments omitted", implies all optional args omitted triggers interactive.
        # But for safety, let's stick to interactive dict structure
    else:
        config = {
            "domain": args.domain or "03_ORCHESTRATION",
            "type": args.type or "IMPLEMENTATION",
            "deliverable": args.deliverable if hasattr(args, 'deliverable') else "Generated Deliverable",
            "requirements": args.requirements if hasattr(args, 'requirements') else "Standard Requirements"
        }

    mission_path = builder.generate_mission_file(config, args.output_dir)
    
    # Queue the mission
    builder.queue_mission(mission_path, args.queue_path, args.priority)
    
    # Validate the mission
    validation_results = builder.validate_mission(mission_path)
    
    if validation_results:
        print(f"Validation results: {validation_results}")
    else:
        print("Mission validated successfully.")
